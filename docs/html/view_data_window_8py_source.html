402 class MyTableModel(QAbstractTableModel):
403  def __init__(self, datain, headerdata, parent=None, *args):
404  QAbstractTableModel.__init__(self, parent, *args)
405  self.arraydata = datain
406  self.headerdata = headerdata
407 
408  def rowCount(self, parent):
409  return len(self.arraydata)
410 
411  def columnCount(self, parent):
412  return len(self.arraydata[0])
413 
414  def data(self, index, role):
415  if not index.isValid():
416  return QVariant()
417  elif role != Qt.DisplayRole:
418  return QVariant()
419  return QVariant(self.arraydata[index.row()][index.column()])
420 
421  def headerData(self, col, orientation, role):
422  if orientation == Qt.Horizontal and role == Qt.DisplayRole:
423  return QVariant(self.headerdata[col])
424  return QVariant()
